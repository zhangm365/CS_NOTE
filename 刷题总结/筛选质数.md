
打表筛选质数 [2,1000001] 以内所有质数




```cpp

#define MAX 1000000+1   // 1e6+1
typedef long long ll;

// 质数数组中只有初始值 2
vector<int> prim = {2};

// 标记数组：seive[i] 是否为质数，避免重复访问
vector<bool> seive(MAX, 0);


void buildPrim()
{
    // 从数值 3 开始枚举
    for( ll i = 3; i < MAX; i += 2 )
    {
        if(!seive[i])
        {
            // 加入质数数组
            prim.emplace_back(i);
            // 标记是 i 倍数的 j 值不是质数
            for( ll j = i*i; j < MAX; j += i )
            {
                seive[j] = true;
            }
        }
    }

}
```
