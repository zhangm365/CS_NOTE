
# 二维数组的前缀和

## 1. C++ 版本

```c++

// 初始二维数组
vector<vector<int>> grid;

int m = grid.size(), n = grid[0].size();

// 二维数组：前缀和
vector<vector<int>> preSum(m+1, vector<int>(n+1, 0));


// 方法1:
for(int i = 1; i <= m; ++i) {
    for(int j = 1; j <= n; ++j) {
        preSum[i][j] = preSum[i][j-1] + grid[i-1][j-1];
    }
}

for(int i = 1; i <= m; ++i) {
    for(int j = 1; j <= n; ++j) {
        preSum[i][j] = preSum[i-1][j] + grid[i-1][j-1];
    }
}



// 方法2
for(int i = 1; i <= m; ++i) {
    for(int j = 1; j <= n; ++j) {
        preSum[i][j] = preSum[i-1][j] + preSum[i][j-1] - preSum[i-1][j-1] 
                        + grid[i-1][j-1];
    }
}


```

## 2. Go 版本

```go

// 创建二维切片
preSum := make([][]int, m)

for i, _ := range preSum {
    preSum[i] = make([]int, n)
}

// 计算二维数组的前缀和
for i := 1; i <= m; i++ {
    for j := 1; j <= n; j++ {
        preSum[i][j] = preSum[i][j-1] + grid[i-1][j-1];
    }
}

for i := 1; i <= m; i++ {
    for j := 1; j <= n; j++ {
        preSum[i][j] = preSum[i-1][j] + grid[i-1][j-1];
    }
}


```
