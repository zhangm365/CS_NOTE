/* 
****中缀表达式转为后缀表达式
**** 2+(3+4)*5
**** 2 3 4 + 5 * +

*/

vector<string> dst;    // 存储后缀表达式

bool isOperator( char c )   // 判断是否为操作符
{
    if( c == '+' || c == '-' || c == '*' || c == '/' )
        return true;
    
    return false;
}

int getPrior( char c )    // 优先级的判断
{
    int level = 0;
    
    switch(c)
    {
        case '(':
            level = 1;
            break;
        case '+':
        case '-':
            level = 2;
            break;
        case '*':
        case '/':
            level = 3;
            break;
        case ')':
            level = 4;
            break;
        default:
            break;
    }
    
    return level;
}

void inToRPN( const string &src )    // 中缀-->后缀
{
    
    int len = src.size();
    int num = 0;
    int i = 0;
    stack<char> op;    // 字符栈
    
    while( i < len )
    {
        if( isdigit(src[i]) )    // 数字
        {
            num = 0;
            do{
                num = num*10 + (src[i] - '0');
                i++;
            }while(i < len && isdigit(src[i]));
            
            dst.push_back(to_string(num));
            
        }
        else if( src[i] == '(' )
        {
            op.push(src[i]);
            i++;
        }
        else if( isOperator(src[i]) )
        {
            if( op.empty() )
            {
                op.push(src[i]);
                i++;
            }
            else
            {
                while( !op.empty() )
                {
                    char c = op.top();
                    if( getPrior(c) >= getPrior(src[i]) )
                    {
                        string tmp = " ";
                        tmp[0] = c;
                        dst.push_back(tmp);
                        op.pop();
                    }
                    else
                    {
                        break;
                    }
                }
                op.push(src[i]);
                i++;
            }
        }
        else if( src[i] == ')' )
        {
            while( op.top() != '(' )    // 一直匹配到 '('
            {
                string tmp = " ";
                tmp[0] = op.top();
                dst.push_back(tmp);
                op.pop();
            }
            op.pop();   // 将 '(' 出栈
            i++;
        }
        else
        {
            i++;
        }
        
    }
    
    while( !op.empty() )
    {
        string tmp = " ";
        tmp[0] = op.top();
        dst.push_back(tmp);
        op.pop();
    }
        
}
